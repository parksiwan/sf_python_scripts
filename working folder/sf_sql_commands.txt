[Insert Monthly Usage]
insert into inventory_monthlyusage (sf_code, usage_month, monthly_usage, unit, product_type)
select sf_code, date_trunc('month', update_date), sum(pickup_qty), unit, product_type from inventory_usage where update_date > '2019-09-30' group by sf_code, date_trunc('month', update_date), unit, product_type order by sf_code, date_trunc('month', update_date)

[Insert Current Stock]
insert into inventory_currentstock (sf_code, end_stock_day, new_balance, unit, bbd, product_type)
select sf_code, update_date, sum(new_balance), unit, bbd, product_type from inventory_stock where update_date = '2019-10-31' group by sf_code, update_date, unit, bbd, product_type order by sf_code, update_date, bbd

[Retrieve Current Stock file]
select * from inventory_currentstock where end_stock_day = '2019-07-11' order by sf_code, bbd

[Retrieve Monthly Usage file]
select * from inventory_monthlyusage order by sf_code, usage_month

[Retrieve Code list file]
(select distinct(sf_code) from inventory_currentstock where end_stock_day = '2019-09-30' 
union
(select distinct(sf_code) from inventory_monthlyusage) order by sf_code
